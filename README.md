};

const app = initializeApp(firebaseConfig);

const db = getFirestore(app);

const postsContainer =
document.getElementById("posts");

/* LOAD POSTS */

function loadPosts(){

const cachedPosts =
localStorage.getItem("factzone_posts");

if(cachedPosts){
postsContainer.innerHTML = cachedPosts;
}

const q =
query(
collection(db,"posts"),
orderBy("createdAt","desc")
);

onSnapshot(q,(querySnapshot)=>{

postsContainer.innerHTML = "";

if(querySnapshot.empty){

postsContainer.innerHTML = `
<h2 class="text-center text-2xl col-span-full">
No Posts Found
</h2>
`;

return;

}

const posts = [];

querySnapshot.forEach((doc)=>{

posts.push({
id:doc.id,
...doc.data()
});

});

posts.sort((a,b)=>
(b.createdAt || 0) - (a.createdAt || 0)
);

posts.forEach((post)=>{

postsContainer.innerHTML += `

<a href="detail.html?id=${post.id}"
class="card shadow-lg">

<img
src="${post.image || 'https://placeholder.com'}"
alt="${post.title || ''}"
class="post-image"
loading="lazy">

<div class="p-4 flex flex-col flex-1">

<h2 class="font-bold text-xl mb-2 line2">
${post.title || ''}
</h2>

<p class="text-gray-600 text-sm line2">
${post.desc || ''}
</p>

</div>

</a>

`;

});

localStorage.setItem(
"factzone_posts",
postsContainer.innerHTML
);

});

}

loadPosts();

/* SEARCH */

document.getElementById("searchInput")
.addEventListener("input",(e)=>{

const value =
e.target.value.toLowerCase();

document.querySelectorAll("#posts a")
.forEach(card=>{

card.style.display =
card.innerText.toLowerCase()
.includes(value)
? "block"
: "none";

});

});
